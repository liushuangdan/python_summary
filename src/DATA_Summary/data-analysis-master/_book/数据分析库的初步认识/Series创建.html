创建

列表创建 Series

#默认索引
a = pd.Series([3,4,5,6])
a

    0    3
    1    4
    2    5
    3    6
    dtype: int64

#自定义索引
b = pd.Series([3,4,5,6],index = ['a','b','c','d'])
b

    a    3
    b    4
    c    5
    d    6
    dtype: int64

# 值可以存储不同类型

c = pd.Series(['小明',18,177,True],index = ['name','age','height','gender'])
c

    name        小明
    age         18
    height     177
    gender    True
    dtype: object

字典创建Series

d = pd.Series({'name':'小明','age':18,'height':177,'gender':True})
d

    name        小明
    age         18
    height     177
    gender    True
    dtype: object

# 覆盖自带的索引

d = pd.Series(
    {'name':'小明','age':18,'height':177,'gender':True},
    index = ['name','age','height','address']
)

d

    name        小明
    age         18
    height     177
    address    NaN
    dtype: object
